float Gain_voltagedb(float Av);

float ES, Vin, Vo, Vnl, Zin, Zo, Rx, Rl, Av, AvdB; 
	
	Rx=  Ask_For_Resx();

    	ES=  Ask_For_Voltage_Source();

float Ask_For_Resx()

}
float Ask_For_Resl()

float Ask_For_Voltage_Source()

	float Es;

float Zinput(float Vin, float ES, float Rx)
{	
	float Zin;
	Zin= (Vin /(ES-Vin))*Rx;
	cout.precision(3);
	return(Zin);
}
float Zout(float Vnl, float Vo, float Rl)
{	float Zo;
	Zo= ((Vnl - Vo)/ Vo)*Rl;
	cout.precision(3);
	return(Zo);
}
float Gain_voltage(float Vo, float Vin)
{	
	float Av;
	Av= Vo/Vin;
	cout.precision(3);
	return(Av);
}
float Gain_voltagedb(float Av)
{
	float AvdB;
	AvdB= 20*log10(Av);
	cout.precision(3);
	return (AvdB);
}
